  \ifnum\@multicnt=\@ne\@firstofone{&\omit}\fi
  \@multicnt#2%
  \advance\@multicnt-#1%
  \advance\@multispan\@ne

  \cr
  \noalign{\nobreak\vskip-\arrayrulewidth}}
